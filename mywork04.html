function fibIterate(n){
	var i, temp0, temp1=0, temp2=1;
	if(n<1){
		return n;
	}
	else{
		for(var i=1; i<=n; i++){
			temp0 = temp1 + temp2;
			temp2 = temp1;
			temp1 = temp0;
		}
